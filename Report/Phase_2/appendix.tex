\section{Uncertainty Scoring Algorithm}
\label{app:uncertainty-alg}
%% ====================================================================

Algorithm~\ref{alg:app-uncertainty} formalizes the four-factor uncertainty scoring described in Section~\ref{sec:sast}. Given a SAST finding with confidence, taint-flow length, interprocedural depth, CWE identifier, severity, and tool verdicts, the algorithm produces a composite uncertainty score $U \in [0, 1]$ that drives escalation routing.

\begin{algorithm}[H]
\caption{Four-Factor Uncertainty Scoring}
\label{alg:app-uncertainty}
\begin{algorithmic}[1]
\Require Finding $f$ with fields: $c_{\text{sast}}$, $l_{\text{taint}}$, $d_{\text{interproc}}$, $\text{cwe\_id}$, severity, tool\_verdicts
\Ensure $U \in [0, 1]$
\State $U_{\text{conf}} \gets \text{clamp}(1.0 - c_{\text{sast}})$
\State $\text{hop\_score} \gets \text{clamp}((l_{\text{taint}} - 1) / 4)$
\State $\text{depth\_score} \gets \text{clamp}(d_{\text{interproc}} / 5)$
\State $U_{\text{comp}} \gets (\text{hop\_score} + \text{depth\_score}) / 2$
\If{$\text{cwe\_id} \in \text{COMMON\_CWE\_IDS}$ (25 entries)}
    \State $U_{\text{nov}} \gets 0.15$
\ElsIf{$\text{cwe\_id}$ is present}
    \State $U_{\text{nov}} \gets 0.85$
\Else
    \State $U_{\text{nov}} \gets 1.0$
\EndIf
\State $U_{\text{confl}} \gets \Call{ConflictScore}{\text{tool\_verdicts}}$ \Comment{0.0/0.1/0.5/1.0}
\State $\delta_{\text{sev}} \gets \text{SEVERITY\_ADJ}[\text{severity}]$ \Comment{+0.15/+0.10/0.00/$-$0.05}
\State $U \gets \text{clamp}(0.4 \cdot U_{\text{conf}} + 0.3 \cdot U_{\text{comp}} + 0.2 \cdot U_{\text{nov}} + 0.1 \cdot U_{\text{confl}} + \delta_{\text{sev}})$
\State \Return $U$
\end{algorithmic}
\end{algorithm}

\section{Conformal Prediction Algorithm}
\label{app:conformal-alg}
%% ====================================================================

Algorithm~\ref{alg:app-aps} details the Adaptive Prediction Sets (APS) procedure described in Section~\ref{sec:graph}. Calibration computes a quantile threshold from held-out nonconformity scores; inference builds a prediction set by greedily including classes until the cumulative probability meets the threshold.

\begin{algorithm}[H]
\caption{APS Calibration and Inference}
\label{alg:app-aps}
\begin{algorithmic}[1]
\Statex \textbf{Calibration:}
\Require Calibration set $\{(\mathbf{x}_i, y_i)\}_{i=1}^{n}$, significance level $\alpha = 0.1$, temperature $T$
\For{$i = 1, \ldots, n$}
    \State $\hat{\boldsymbol{\pi}}_i \gets \text{softmax}(\mathbf{z}_i / T)$ \Comment{Temperature-scaled probabilities}
    \State Sort classes by descending $\hat{\pi}_{i,j}$
    \State $s_i \gets \text{CumulativeSum at rank of } y_i$
\EndFor
\State $\hat{q} \gets \text{Quantile}(\{s_1, \ldots, s_n\}, \lceil(n+1)(1-\alpha)\rceil / n, \text{method=``higher''})$

\Statex
\Statex \textbf{Inference:}
\Require New sample $\mathbf{x}$, threshold $\hat{q}$
\State $\hat{\boldsymbol{\pi}} \gets \text{softmax}(\mathbf{z} / T)$
\State Sort classes by descending $\hat{\pi}_j$
\State $\mathcal{C} \gets \emptyset$
\For{$j = 1, 2, \ldots$ in sorted order}
    \State $\mathcal{C} \gets \mathcal{C} \cup \{\text{class}_j\}$
    \If{$\sum_{k \in \mathcal{C}} \hat{\pi}_k \geq \hat{q}$}
        \State \textbf{break}
    \EndIf
\EndFor
\State \Return $\mathcal{C}$, $1 - \alpha$
\end{algorithmic}
\end{algorithm}

\section{Consensus Protocol Algorithm}
\label{app:consensus-alg}
%% ====================================================================

Algorithm~\ref{alg:app-consensus} specifies the five-rule dual-agent consensus protocol described in Section~\ref{sec:llm}. The protocol combines attacker exploitability evidence with defender coverage metrics to produce a final verdict and confidence score.

\begin{algorithm}[H]
\caption{Dual-Agent Consensus Protocol}
\label{alg:app-consensus}
\begin{algorithmic}[1]
\Require AttackerVerdict $(e, c_{\text{atk}}, \text{payload}, \text{AV}, \text{AC}, \text{PR}, \text{UI})$
\Require DefenderVerdict $(d_{\text{cov}}, p_{\text{feas}}, \text{sanitizers}, \text{S}, \text{C}, \text{I}, \text{A})$
\Require Thresholds: $\tau_{\text{conf}} = 0.5$, $\tau_{\text{safe}} = 0.7$, $c_{\text{infeas}} = 0.8$
\If{$e$ \textbf{and} $d_{\text{cov}} < \tau_{\text{conf}}$} \Comment{Rule 1: Strong vulnerability evidence}
    \State \Return (CONFIRMED, $\max(c_{\text{atk}}, 1 - d_{\text{cov}})$)
\ElsIf{$\neg e$ \textbf{and} $d_{\text{cov}} > \tau_{\text{safe}}$} \Comment{Rule 2: Strong safety evidence}
    \State \Return (SAFE, $\max(d_{\text{cov}}, 1 - c_{\text{atk}})$)
\ElsIf{$\neg e$ \textbf{and} $\neg p_{\text{feas}}$} \Comment{Rule 2b: Path infeasible}
    \State \Return (SAFE, $c_{\text{infeas}}$)
\ElsIf{$e$ \textbf{and} $d_{\text{cov}} \geq \tau_{\text{conf}}$} \Comment{Rule 3: Exploitable but defended}
    \State $c \gets \text{clamp}(0.5 + 0.3(c_{\text{atk}} - d_{\text{cov}}),\; 0.3,\; 0.85)$
    \State \Return (LIKELY, $c$)
\Else \Comment{Rule 4: Not exploitable, weak defense}
    \State $c \gets \text{clamp}(0.4 + 0.2(1 - d_{\text{cov}}),\; 0.2,\; 0.6)$
    \State \Return (POTENTIAL, $c$)
\EndIf
\end{algorithmic}
\end{algorithm}
